代码示例如下：::

    num = 20
    ndim = 2
    x = np.ones((num, ndim))

    x[:, 0] = np.linspace(-10, 10.0, num)
    x[:, 1] = 0.0
    s = Sphere()
    dataSet = s(x)
    plt.plot(dataSet["input"][:, 0], dataSet["output"][:, 0])
    plt.xlabel("x")
    plt.ylabel("y")
    plt.show()
    print(dataSet)

    know = s.getKnowledge(knowType=["单调型","形状型"],visual = True)
    print(know)
